Scoping - IIFE

(function (professor) {

  var name = 'Powell';

  professor.favColor = 'Green';
  professor.sayHi = function () {
    console.log(`Prof ${name} says hi everybody!`);
  };

})(window.professor = window.professor || {});
Variable Accessible Outside IIFE
window.professor No
window.professor.favColor No
window.professor.sayHi() No
window.name Only through sayHi()
